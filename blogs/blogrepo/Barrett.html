#define ull unsigned long long 
#define ui128 __uint128_t
struct Barrett {
    ull d;
    ui128 m;
    Barrett(ull _d):d(_d),m((((ui128)(1)<<64)/d)) {}
    ull operator ()(ull a) {
        ull w=(m*a)>>64;
        w=a-w*d;
        if(w>=d) w-=d;
        return w;
    }
};
