int exgcd(int a,int b,int &x,int &y) {
    x=1,y=0;
    int n=0,m=1,t;
    while(b) {
        t=n,n=x-a/b*n,x=t;
        t=m,m=y-a/b*m,y=t;
        t=b,b=a%b,a=t;
    }
    return a;
}
